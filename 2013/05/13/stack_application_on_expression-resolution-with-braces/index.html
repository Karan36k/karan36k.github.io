Solving operation expressions with brackets-basic application of stack 3

Posted on 2013-05-13 Edited on 2020-04-09 In Data Structure Views:
Symbols count in article: 5.8k Reading time ≈ 5 mins.

This article is based on the program written by Yan Weimin’s data structure (c language version). If necessary, go to the video first. If there are any mistakes, please welcome them to avoid harming others.

Examples

Exp = a * b + (c – d / e) * f

Prefix: + * ab *-c / def

Infix: a * b + c–d / e * f

Suffix: ab * cd e / -f * +

Same point

The numbers are arranged in the original formula: so the operands are stacked on the stack in order.

difference

1: The order of operators in the suffix is ​​exactly the order of solving

2: Each operator and the 2 operands that appear immediately before it and form a minimum expression

The key: to find the suffix from the original expression

Application steps

-Step1: Set up two stacks, one operator stack and the other suffix stack.
-Step2: Add the = sign before and after the expression to indicate the start and end of the operation expression, so in the operator, the = sign has the lowest priority.
-Step3: If the current character is an operand, it is sent directly to the postfix stack. Consistent with the common point mentioned above: the numbers are in the order of the original expression from left to right.
-Step4: The priority of the left parenthesis is higher than the operator before the left parenthesis, and the operator after the left parenthesis is higher than the left parenthesis, so as to play a role in isolation, the operator before the right parenthesis is higher than the right parenthesis, Only in this way can the parentheses isolate the inner expression!
-Step5: If the priority of the current operator is higher than the operator at the top of the stack, the operator stack is entered, otherwise the operator at the top of the stack that exits the operator stack and the result of the two operands taken from the top of the operand stack are used as The new operand is pushed onto the operand stack, and then the current operator is pushed onto the operator stack.

#include <stdio.h>

#include <malloc.h> // malloc ()
#include <process.h> // exit ();
// Function result status code
#define TRUE 1
#define FALSE 0
#define OK 1
#define ERROR 0
#define INFEASIBLE -1

#define STACK_INIT_SIZE 100 // Initial allocation of storage space
#define STACKINCREMENT 2 // Storage space allocation increment

typedef int Status; // Status is the type of the function, its value is the function result status code, such as OK, etc.
typedef char SElemType;
struct SqStack
{
SElemType * base; // Before stack construction and after destruction, the value of base is NULL
SElemType * top; // pointer to the top of the stack
int stacksize; // currently allocated storage space, in elements
};

Status InitStack (SqStack & S)
{// Construct an empty stack S
if (! (S.base = (SElemType *) malloc (STACK_INIT_SIZE * sizeof (SElemType))))
exit (-1); // storage allocation failed
S.top = S.base;
S.stacksize = STACK_INIT_SIZE;
return OK;
}

Status DestroyStack (SqStack & S)
{// Destroy stack S, S no longer exists
free (S.base);
S.base = NULL;
S.top = NULL;
S.stacksize = 0;
return OK;
}

Status ClearStack (SqStack & S)
{// Set S to the empty stack
S.top = S.base;
return OK;
}

Status StackEmpty (SqStack S)
{// If stack S is empty, then return TRUE, otherwise return FALSE
if (S.top == S.base)
return TRUE;
else
return FALSE;
}

int StackLength (SqStack S)
{// Return the number of elements of S, that is, the length of the stack
return S.top-S.base;
}

Status GetTop (SqStack S, SElemType & e)
{// If the stack is not empty, use e to return the top element of S, and return OK; otherwise return ERROR
if (S.top> S.base)
{
e = * (S.top-1);
return OK;
}
else
return ERROR;
}

Status Push (SqStack & S, SElemType e)
{// Insert element e as the new top element
if (S.top-S.base> = S.stacksize) // Stack is full, additional storage space
{
S.base = (SElemType *) realloc (S.base, (S.stacksize + STACKINCREMENT) * sizeof (SElemType));
if (! S.base)
exit (-1); // storage allocation failed
S.top = S.base + S.stacksize;
S.stacksize + = STACKINCREMENT;
}
* (S.top) ++ = e;
return OK;
}

Status Pop (SqStack & S, SElemType & e)
{// If the stack is not empty, delete the top element of S, return its value with e, and return OK; otherwise return ERROR
if (S.top == S.base)
return ERROR;
e = *-S.top;
return OK;
}

Status StackTraverse (SqStack S, Status (* visit) (SElemType))
{// Call the function visit () for each element in the stack from the bottom to the top of the stack.
// Once visit () fails, the operation fails
while (S.top> S.base)
visit (* S.base ++);
printf (“\ n”);
return OK;
}

SElemType Precede (SElemType a, SElemType b)
{// judgment operator priority
int i, j;
char Table [8] [8] = {
{‘’, ‘+’, ‘-‘, ‘‘, ‘/‘, ‘(‘, ‘)’, ‘#’},
{‘+’, ‘>’, ‘>’, ‘<’, ‘<’, ‘<’, ‘>’, ‘>’},
{‘-‘, ‘>’, ‘>’, ‘<’, ‘<’, ‘<’, ‘>’, ‘>’},
{‘‘, ‘>’, ‘>’, ‘>’, ‘>’, ‘<’, ‘>’, ‘>’},
{‘/‘, ‘>’, ‘>’, ‘>’, ‘>’, ‘<’, ‘>’, ‘>’},
{‘(‘, ‘<’, ‘<’, ‘<’, ‘<’, ‘<’, ‘=’, ‘’},
{‘)’, ‘>’, ‘>’, ‘>’, ‘>’, ‘’, ‘>’, ‘>’},
{‘#’, ‘<’, ‘<’, ‘<’, ‘<’, ‘<’, ‘’, ‘=’}
}; // Priority table

for (i = 0; i <8; i ++)
    if (Table [0] [i] == a) // Find operator a
        break;
for (j = 0; j <8; j ++) // find operator
    if (Table [j] [0] == b)
        break;
return Table [j] [i];

}

Status In (SElemType c)
{// Determine whether c is an operator
switch (c)
{
case ‘+’:
case’- ‘:
case ‘*’:
case ‘/‘:
case ‘(‘:
case ‘)’:
case ‘#’: return TRUE;
default: return FALSE;
}
}

SElemType Operate (SElemType a, SElemType theta, SElemType b)
{
SElemType c;
a = a-48;
b = b-48;
switch (theta)
{
case ‘+’: c = a + b + 48;
break;
case ‘-‘: c = a-b + 48;
break;
case ‘*’: c = a * b + 48;
break;
case ‘/‘: c = a / b + 48;
}
return c;
}

SElemType EvaluateExpression () // Algorithm 3.4
{// Operator-first algorithm for arithmetic expression evaluation. Let OPTR and OPND be the operator stack and operand stack, respectively
SqStack OPTR, OPND;
SElemType a, b, c, x, theta;
InitStack (OPTR);
Push (OPTR, ‘#’);
InitStack (OPND);
c = getchar ();
GetTop (OPTR, x);
while (c! = ‘#’ || x! = ‘#’)
{
if (In (c)) // is one of 7 operators
switch (Precede (c, x))
{
case ‘<’: Push (OPTR, c); // Low priority of the top element of the stack
c = getchar ();
break;
case ‘=’: Pop (OPTR, x); // Remove parentheses and receive the next character
c = getchar ();
break;
case ‘>’: Pop (OPTR, theta); // Unstack and put the operation result on the stack
Pop (OPND, b);
Pop (OPND, a);
Push (OPND, Operate (a, theta, b));
break;
}
else if (c> = ‘0’ && c <= ‘9’) // c is the operand
{
Push (OPND, c);
c = getchar ();
}
else // c is an illegal character
{
printf (“Illegal character \ n”);
exit (-1);
}
GetTop (OPTR, x);
}
GetTop (OPND, x);
return x;
}

int main ()
{
printf (“Please enter the arithmetic expression (the intermediate value and the final result should be between 0-9), and end with # \“ “;
printf (“% c \ n”, EvaluateExpression ());
return 0;
}
